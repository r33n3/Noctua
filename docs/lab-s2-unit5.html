def investigate(alert_text: str, title: str = "SOC Investigation") -> str:
    session = client.beta.sessions.create(
        agent=AGENT_ID,
        environment_id=ENV_ID,
        title=title,
    )
    print(f"Session: {session.id}\n")

    output = []
    # Open stream BEFORE sending — avoids race condition
    with client.beta.sessions.events.stream(session.id) as stream:
        client.beta.sessions.events.send(
            session.id,
            events=[{"type": "user.message", "content": [
                {"type": "text", "text": alert_text}
            ]}],
        )
        for event in stream:
            if event.type == "agent.message":
                for block in event.content:
                    if hasattr(block, "text"):
                        print(block.text, end="", flush=True)
                        output.append(block.text)
            elif event.type == "agent.tool_use":
                print(f"\n[Tool: {event.name}]", flush=True)
            elif event.type == "session.status_idle":
                print("\n\n── Investigation complete ──")
                break

    return "".join(output)

A2A vs. MCP — Two Protocols, Two Jobs

Two protocols appear in agentic system architecture. They solve different problems:

  • MCP (Model Context Protocol) — how an agent calls a tool. Defines the interface between an agent and external capabilities: APIs, databases, file systems. You built an MCP server in S1 Unit 2.
  • A2A (Agent-to-Agent Protocol) — how agents communicate with each other. Defines message passing, task delegation, and result handoff between agents in a multi-agent system.

A production multi-agent system needs both: MCP to connect agents to tools, A2A to connect agents to each other. In Managed Agents, callable_agents[] handles A2A at one delegation level (coordinator → subagent). MCP is how each agent accesses tools — yours from Unit 2 included.

# runner_with_custom_tool.py — handle agent.custom_tool_use events
import anthropic, os, json, subprocess

client = anthropic.Anthropic()

MCP_SERVER_URL = "http://localhost:3000"

def call_mcp_server(tool_name: str, arguments: dict) -> str:
    """Call your local MCP server and return the result as a string."""
    import urllib.request
    payload = json.dumps({"tool": tool_name, "arguments": arguments}).encode()
    req = urllib.request.Request(
        f"{MCP_SERVER_URL}/call",
        data=payload,
        headers={"Content-Type": "application/json"},
    )
    with urllib.request.urlopen(req, timeout=10) as resp:
        return resp.read().decode()

def investigate_with_mcp(alert_text: str) -> str:
    session = client.beta.sessions.create(
        agent=os.environ["AGENT_V2_ID"],
        environment_id=os.environ["ENVIRONMENT_ID_PROD"],
        resources=[{
            "type": "memory_store",
            "memory_store_id": os.environ["MEMORY_STORE_ID"],
            "access": "read_write",
            "instructions": "Check /context/ for thresholds. Write findings to /findings/.",
        }],
    )

    events_by_id = {}
    report_parts = []

    with client.beta.sessions.events.stream(session.id) as stream:
        client.beta.sessions.events.send(
            session.id,
            events=[{"type": "user.message", "content": [
                {"type": "text", "text": alert_text}
            ]}],
        )
        for event in stream:
            events_by_id[event.id] = event

            if event.type == "agent.message":
                for block in event.content:
                    if hasattr(block, "text"):
                        print(block.text, end="", flush=True)
                        report_parts.append(block.text)
            elif event.type == "agent.custom_tool_use":
                print(f"\n[Custom tool: {event.name}({event.input})]", flush=True)
            elif event.type == "session.status_idle":
                stop = event.stop_reason
                if stop and stop.type == "requires_action":
                    # Agent called a custom tool — execute locally and return result
                    for event_id in stop.event_ids:
                        tool_event = events_by_id[event_id]
                        result = call_mcp_server(tool_event.name, tool_event.input)
                        client.beta.sessions.events.send(
                            session.id,
                            events=[{
                                "type": "user.custom_tool_result",
                                "custom_tool_use_id": event_id,
                                "content": [{"type": "text", "text": result}],
                            }],
                        )
                elif stop and stop.type == "end_turn":
                    print("\n── Investigation complete ──")
                    break

    return "".join(report_parts)
Architecture note: Your local runner process is the bridge. Anthropic's cloud calls your custom tool; your code executes it against localhost:3000 and returns the result. The session's allow_mcp_servers: true flag in the environment is for MCP servers reachable directly from the container — a different pattern for publicly-accessible MCP endpoints.
